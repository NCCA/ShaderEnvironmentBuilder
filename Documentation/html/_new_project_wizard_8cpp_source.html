52 void NewProjectWizard::accept()
53 {
54  // get project name and directory
55  m_output->m_projectName = field("projectName").toString().toStdString();
56  m_output->m_projectDir = field("projectDirectory").toString().toStdString();
57 
58  // create glsl version string
59  QString ver = QString("#version %1 %2\n\n")
60  .arg(field("glslVersion").toString(),
61  field("glslProfile").toString());
62 
63  // get the selected rows from the vertex and fragment selection models used
64  // with the treeviews
65  QModelIndexList vertFiles = m_vertexSelectModel->selectedRows();
66  QModelIndexList fragFiles = m_fragmentSelectModel->selectedRows();
67 
68  // Get the glsl order page
69  GlslOrderPage* glslOrderPg = static_cast<GlslOrderPage*>(page(3));
70 
71  // Setup variables for getting the correct order of the files
72  // ordered filenames (from the qlistwidgets)
73  QStringList vertexOrderFileNames, fragmentOrderFileNames;
74  // unordered filenames (from the qtreeviews)
75  QStringList vertFileNames, fragFileNames;
76 
77  // get the list of ordered files from the list widgets
78  const QListWidget* vertFilesOrder = glslOrderPg->getVertListWidget();
79  const QListWidget* fragFilesOrder = glslOrderPg->getFragListWidget();
80 
81  // create the unordered and ordered file name lists for vertex
82  for (int i=0; i<vertFilesOrder->count(); ++i)
83  {
84  vertFileNames.append(m_fileModel->fileInfo(vertFiles.at(i)).fileName());
85  vertexOrderFileNames.append(vertFilesOrder->item(i)->text());
86  }
87 
88  // create the unordered and ordered file name lists for fragment
89  for (int i=0; i<fragFilesOrder->count(); ++i)
90  {
91  fragFileNames.append(m_fileModel->fileInfo(fragFiles.at(i)).fileName());
92  fragmentOrderFileNames.append(fragFilesOrder->item(i)->text());
93  }
94 
95  // the final vertex and fragment strings
96  QString vertexFilesString, fragmentFilesString;
97 
98  // Now find the correct index from the ordered list using the ordered and
99  // un-ordered file name lists and get the file value from the file model
100  for (int i=0; i<vertFileNames.length(); ++i)
101  {
102  // getting the correct index from the ordered file list
103  int index = vertFileNames.indexOf(vertexOrderFileNames[i]);
104  // get the file at the correct index from the selected rows in the QTreeView
105  QFile vFile(m_fileModel->fileInfo(vertFiles.at(index)).absoluteFilePath());
106  // open the file
107  if (!vFile.open(QIODevice::ReadOnly | QIODevice::Text))
108  {
109  // Raise an error if failed
110  CEBRaise::QtFileError(vFile.error(),
111  m_fileModel->fileInfo(vertFiles.at(index))
112  .absoluteFilePath());
113  }
114  // append the file to a single string
115  QTextStream in(&vFile);
116  vertexFilesString.append(in.readAll());
117  vertexFilesString.append("\n");
118  }
119 
120  // REPEAT above again
121  // Now find the correct index from the ordered list using the ordered and
122  // un-ordered file name lists and get the file value from the file model
123  for (int i=0; i<fragFileNames.length(); ++i)
124  {
125  // getting the correct index from the ordered file list
126  int index = fragFileNames.indexOf(fragmentOrderFileNames[i]);
127  // get the file at the correct index from the selected rows in the QTreeView
128  QFile fFile(m_fileModel->fileInfo(fragFiles.at(index)).absoluteFilePath());
129  // open the file
130  if (!fFile.open(QIODevice::ReadOnly | QIODevice::Text))
131  {
132  // Raise an error if failed
133  CEBRaise::QtFileError(fFile.error(),
134  m_fileModel->fileInfo(fragFiles.at(index))
135  .absoluteFilePath());
136  }
137  // append the file to a single string
138  QTextStream in(&fFile);
139  fragmentFilesString.append(in.readAll());
140  fragmentFilesString.append("\n");
141  }
142 
143  // check if each of the file strings already contains a version. If it doesn't
144  // then push to front
145  if(!vertexFilesString.contains("#version"))
146  {
147  vertexFilesString.push_front(ver+"\n");
148  }
149  // same as above
150  if(!fragmentFilesString.contains("#version"))
151  {
152  fragmentFilesString.push_front(ver+"\n");
153  }
154  // store in output
155  m_output->m_vertSource = vertexFilesString;
156  m_output->m_fragSource = fragmentFilesString;
157 
158  // accept the dialog
159  QDialog::accept();
160 }
